💻 שלב 4: הקוד המלא ומצב בדיקה

העתיקו את הקוד המלא ל-Google Colab. הוספנו אופציה 3 חדשה המאפשרת להעלות את תמונות הבדיקה (ה-20%) ולחשב אוטומטית את אחוזי הדיוק של המערכת. 

import tensorflow as tf
import numpy as np
import cv2
import os
from google.colab import drive, files
from google.colab.patches import cv2_imshow
from IPython.display import display, Javascript, Audio
from google.colab.output import eval_js
from base64 import b64decode

# 1. Mount Google Drive
drive.mount('/content/drive')

# 2. Paths Configuration
PATH = '/content/drive/MyDrive/Gemstone_Project/'
MODEL_PATH = os.path.join(PATH, 'saved_model')
LABELS_PATH = os.path.join(PATH, 'labels.txt')

# 3. Load Model and Labels
if os.path.exists(MODEL_PATH):
    model = tf.keras.models.load_model(MODEL_PATH)
    with open(LABELS_PATH, "r", encoding="utf-8") as f:
        class_names = [line.strip().split(' ', 1)[-1] for line in f.readlines()]
    print("✅ System Ready!")
else:
    print("❌ Error: Path not found!")

def predict_single(image_path, silent=False):
    img = cv2.imread(image_path)
    if img is None: return None, 0
    img_resized = cv2.resize(img, (224, 224))
    img_array = np.asarray(img_resized, dtype=np.float32).reshape(1, 224, 224, 3)
    img_array = (img_array / 127.5) - 1
    prediction = model.predict(img_array, verbose=0)
    index = np.argmax(prediction)
    if not silent:
        print(f"Prediction: {class_names[index]} ({prediction[0][index]*100:.1f}%)")
        cv2_imshow(cv2.resize(img, (200, 150)))
    return class_names[index], prediction[0][index]

# Main Menu
print("\n--- Menu ---")
print("1 - Camera | 2 - Single Upload | 3 - Accuracy Test (Upload Multiple)")
choice = input("Select: ")

if choice == '3':
    print("Upload your 20% test images...")
    uploaded = files.upload()
    correct = 0
    total = len(uploaded)
    
    print("\n--- Starting Statistical Analysis ---")
    for filename in uploaded.keys():
        print(f"\nImage: {filename}")
        print("Available classes:", class_names)
        actual = input(f"What is the CORRECT class for this image? ").strip()
        pred_label, conf = predict_single(filename, silent=False)
        
        if pred_label.lower() == actual.lower():
            correct += 1
            print("✔️ Correct!")
        else:
            print(f"❌ Wrong! (Expected: {actual})")
            
    accuracy = (correct / total) * 100
    print(f"\n============================")
    print(f"FINAL ACCURACY: {accuracy:.2f}%")
    print(f"============================")
    
    if accuracy < 85:
        print("\n⚠️ Accuracy is too low! Suggestions:")
        print("1. Add more diverse images to your training set.")
        print("2. Ensure backgrounds are neutral.")
        print("3. Check for 'Label Noise' (wrongly labeled images).")
